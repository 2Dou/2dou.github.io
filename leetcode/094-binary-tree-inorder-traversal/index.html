class Solution {
public:
    vector<int> inorderTraversal(TreeNode* root) {
        vector<int> r;
        inorderVisit(r, root);
        return r;
    }

private:
    void inorderVisit(vector<int> &r, TreeNode *node) {
        std::stack<TreeNode *> s;
        auto *cur = node;
        while(cur || !s.empty()) {
            while(cur) {
                s.push(cur);
                cur = cur->left;
            }
            cur = s.top();
            s.pop();
            r.push_back(cur->val);
            cur = cur->right;
        }
    }

    /* 递归
    void inorderVisit(vector<int> &r, TreeNode *node) {
        if (!node) {
            return;
        }
        if (node->left) {
            inorderVisit(r, node->left);
        }
        r.push_back(node->val);
        if (node->right) {
            inorderVisit(r, node->right);
        }
    }
    */
};
